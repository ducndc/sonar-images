71 cv::Mat
72 Contours (cv::Mat inputImage)
73 {
74  int thresh = 150;
75  cv::Mat outputImage;
76  cv::Canny (inputImage, outputImage, thresh, 160);
77  std::vector<std::vector<cv::Point>> contours;
78  std::vector<cv::Vec4i> hierarchy;
79  cv::findContours (outputImage, contours, hierarchy, cv::RETR_TREE, cv::CHAIN_APPROX_SIMPLE);
80  cv::Mat drawing = cv::Mat::zeros (outputImage.size (), CV_8UC3);
81  for (size_t i = 0; i < contours.size (); ++i)
82  {
83  cv::Scalar color = cv::Scalar (255, 255, 255);
84  if (cv::contourArea (contours[i]) < 1100)
85  {
86  continue;
87  }
88  cv::drawContours (drawing, contours, (int)i, color, 2, cv::LINE_8, hierarchy, 0);
89  }
90  return drawing;
91 }

168 cv::Mat
169 K_Means (cv::Mat Input, int K)
170 {
171  cv::Mat samples (Input.rows * Input.cols, Input.channels (), CV_32F);
172  for (int y = 0; y < Input.rows; y++)
173  for (int x = 0; x < Input.cols; x++)
174  for (int z = 0; z < Input.channels (); z++)
175  if (Input.channels () == 3)
176  {
177  samples.at<float>(y + x * Input.rows, z) = Input.at<cv::Vec3b>(y, x)[z];
178  }
179  else
180  {
181  samples.at<float>(y + x * Input.rows, z) = Input.at<uchar>(y, x);
182  }
183 
184  cv::Mat labels;
185  int attempts = 5;
186  cv::Mat centers;
187  cv::kmeans (samples, K, labels, cv::TermCriteria(CV_TERMCRIT_ITER | CV_TERMCRIT_EPS, 10000, 0.0001), attempts,
188  cv::KMEANS_PP_CENTERS, centers);
189 
190 
191  cv::Mat new_image (Input.size (), Input.type ());
192  for (int y = 0; y < Input.rows; y++)
193  for (int x = 0; x < Input.cols; x++)
194  {
195  int cluster_idx = labels.at<int>(y + x * Input.rows, 0);
196  if (Input.channels () == 3)
197  {
198  for (int i = 0; i < Input.channels (); i++)
199  {
200  new_image.at<cv::Vec3b>(y, x)[i] = centers.at<float>(cluster_idx, i);
201  }
202  }
203  else
204  {
205  new_image.at<uchar>(y, x) = centers.at<float>(cluster_idx, 0);
206  }
207  }
208  cv::imshow("clustered image", new_image);
209  cv::waitKey(0);
210  return new_image;
211 }
